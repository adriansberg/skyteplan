<!-- Git hash with Vercel fallback -->
try {
  hash = execSync('git rev-parse --short HEAD', { encoding: 'utf-8' }).trim();
} catch {
  const sha = process.env.VERCEL_GIT_COMMIT_SHA ?? 'unknown';
  hash = sha.slice(0, 7);
}
version = 'stordalen-' + hash;
